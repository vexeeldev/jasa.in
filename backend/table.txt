-- =========================
-- USERS
-- =========================
CREATE TABLE USERS (
    user_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    username VARCHAR2(50) NOT NULL UNIQUE,
    email VARCHAR2(100) NOT NULL UNIQUE,
    password_hash VARCHAR2(255) NOT NULL,
    full_name VARCHAR2(100),
    phone VARCHAR2(20),
    avatar_url VARCHAR2(255),
    role VARCHAR2(20) DEFAULT 'klien',
    balance NUMBER(12,2) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT chk_role CHECK (role IN ('klien','freelancer','admin'))
);

-- =========================
-- FREELANCER
-- =========================
CREATE TABLE FREELANCER_PROFILES (
    freelancer_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    user_id NUMBER UNIQUE,
    bio CLOB,
    rating_avg NUMBER(3,2) DEFAULT 0,
    total_orders NUMBER DEFAULT 0,
    freelancer_level VARCHAR2(20) DEFAULT 'new',
    joined_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES USERS(user_id) ON DELETE CASCADE
);

-- =========================
-- SKILLS
-- =========================
CREATE TABLE SKILLS (
    skill_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    name VARCHAR2(100) UNIQUE,
    category VARCHAR2(100)
);

CREATE TABLE FREELANCER_SKILLS (
    freelancer_id NUMBER,
    skill_id NUMBER,
    PRIMARY KEY (freelancer_id, skill_id),
    FOREIGN KEY (freelancer_id) REFERENCES FREELANCER_PROFILES(freelancer_id) ON DELETE CASCADE,
    FOREIGN KEY (skill_id) REFERENCES SKILLS(skill_id) ON DELETE CASCADE
);

-- =========================
-- CATEGORIES
-- =========================
CREATE TABLE CATEGORIES (
    category_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    parent_id NUMBER,
    name VARCHAR2(100),
    slug VARCHAR2(100) UNIQUE,
    icon_url VARCHAR2(255),
    FOREIGN KEY (parent_id) REFERENCES CATEGORIES(category_id)
);

-- =========================
-- SERVICES
-- =========================
CREATE TABLE SERVICES (
    service_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    freelancer_id NUMBER,
    category_id NUMBER,
    title VARCHAR2(200),
    description CLOB,
    thumbnail_url VARCHAR2(255),
    status VARCHAR2(20) DEFAULT 'active',
    total_orders NUMBER DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (freelancer_id) REFERENCES FREELANCER_PROFILES(freelancer_id),
    FOREIGN KEY (category_id) REFERENCES CATEGORIES(category_id)
);

CREATE TABLE SERVICE_PACKAGES (
    package_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    service_id NUMBER,
    package_name VARCHAR2(20),
    description CLOB,
    price NUMBER(12,2),
    delivery_days NUMBER,
    revisions NUMBER,
    FOREIGN KEY (service_id) REFERENCES SERVICES(service_id) ON DELETE CASCADE
);

-- =========================
-- PORTFOLIO
-- =========================
CREATE TABLE PORTFOLIOS (
    portfolio_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    freelancer_id NUMBER,
    title VARCHAR2(200),
    description CLOB,
    image_url VARCHAR2(255),
    project_url VARCHAR2(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (freelancer_id) REFERENCES FREELANCER_PROFILES(freelancer_id)
);

-- =========================
-- ORDERS SYSTEM
-- =========================
CREATE TABLE ORDERS (
    order_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    package_id NUMBER,
    client_id NUMBER,
    freelancer_id NUMBER,
    status VARCHAR2(20) DEFAULT 'pending',
    total_price NUMBER(12,2),
    deadline DATE,
    requirements CLOB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP,
    FOREIGN KEY (package_id) REFERENCES SERVICE_PACKAGES(package_id),
    FOREIGN KEY (client_id) REFERENCES USERS(user_id),
    FOREIGN KEY (freelancer_id) REFERENCES FREELANCER_PROFILES(freelancer_id)
);

CREATE TABLE ORDER_REVISIONS (
    revision_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    order_id NUMBER,
    description CLOB,
    attachment_url VARCHAR2(255),
    requested_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (order_id) REFERENCES ORDERS(order_id)
);

CREATE TABLE REVIEWS (
    review_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    order_id NUMBER UNIQUE,
    reviewer_id NUMBER,
    rating NUMBER(1),
    review_comment CLOB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (order_id) REFERENCES ORDERS(order_id),
    FOREIGN KEY (reviewer_id) REFERENCES USERS(user_id),
    CONSTRAINT chk_rating CHECK (rating BETWEEN 1 AND 5)
);

-- =========================
-- PAYMENTS
-- =========================
CREATE TABLE PAYMENTS (
    payment_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    order_id NUMBER UNIQUE,
    amount NUMBER(12,2),
    method VARCHAR2(50),
    status VARCHAR2(20),
    transaction_ref VARCHAR2(100),
    paid_at TIMESTAMP,
    FOREIGN KEY (order_id) REFERENCES ORDERS(order_id)
);

-- =========================
-- LEDGER SYSTEM (FINANCE CORE)
-- =========================
CREATE TABLE TRANSACTIONS (
    transaction_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    user_id NUMBER,
    type VARCHAR2(10), -- credit / debit
    amount NUMBER(12,2),
    reference_id NUMBER,
    reference_type VARCHAR2(20),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES USERS(user_id)
);

CREATE TABLE TOPUPS (
    topup_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    user_id NUMBER,
    amount NUMBER(12,2),
    payment_method VARCHAR2(50),
    status VARCHAR2(20),
    transaction_ref VARCHAR2(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    paid_at TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES USERS(user_id)
);

CREATE TABLE WITHDRAWALS (
    withdrawal_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    freelancer_id NUMBER,
    amount NUMBER(12,2),
    bank_name VARCHAR2(50),
    account_number VARCHAR2(50),
    status VARCHAR2(20),
    requested_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    processed_at TIMESTAMP,
    FOREIGN KEY (freelancer_id) REFERENCES FREELANCER_PROFILES(freelancer_id)
);

-- =========================
-- MESSAGING
-- =========================
CREATE TABLE MESSAGES (
    message_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    sender_id NUMBER,
    receiver_id NUMBER,
    order_id NUMBER,
    content CLOB,
    is_read CHAR(1) DEFAULT '0',
    sent_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (sender_id) REFERENCES USERS(user_id),
    FOREIGN KEY (receiver_id) REFERENCES USERS(user_id),
    FOREIGN KEY (order_id) REFERENCES ORDERS(order_id)
);

-- =========================
-- NOTIFICATIONS
-- =========================
CREATE TABLE NOTIFICATIONS (
    notif_id NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    user_id NUMBER,
    type VARCHAR2(50),
    title VARCHAR2(200),
    body CLOB,
    is_read CHAR(1) DEFAULT '0',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES USERS(user_id)
);
